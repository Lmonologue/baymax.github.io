这里给出使用opencv时，编译main.cpp的makefile文件，要注意`-L. -lmylib`的位置
```
CFLAGS=-g -Wall
OPENCVLIB=-lopencv_core -lopencv_highgui -lopencv_imgproc -lpthread
OBJS=main.cpp
CXX=g++
all: demo
demo: $(OBJS)
	$(CXX) $(CFLAGS) $(OBJS) -L. -lmylib $(OPENCVLIB) -o $@ 
```
